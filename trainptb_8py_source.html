44 # Dataset iterator to create a batch of sequences at different positions.
45 # This iterator returns a pair of current words and the next words. Each
46 # example is a part of sequences starting from the different offsets
47 # equally spaced within the whole sequence.
48 class ParallelSequentialIterator(chainer.dataset.Iterator):
49 
50  def __init__(self, dataset, batch_size, repeat=True):
51  self.dataset = dataset
52  self.batch_size = batch_size # batch size
53  # Number of completed sweeps over the dataset. In this case, it is
54  # incremented if every word is visited at least once after the last
55  # increment.
56  self.epoch = 0
57  # True if the epoch is incremented at the last iteration.
58  self.is_new_epoch = False
59  self.repeat = repeat
60  length = len(dataset)
61  # Offsets maintain the position of each sequence in the mini-batch.
62  self.offsets = [i * length // batch_size for i in range(batch_size)]
63  # NOTE: this is not a count of parameter updates. It is just a count of
64  # calls of ``__next__``.
65  self.iteration = 0
66  # use -1 instead of None internally
67  self._previous_epoch_detail = -1.
68 
69  def __next__(self):
70  # This iterator returns a list representing a mini-batch. Each item
71  # indicates a different position in the original sequence. Each item is
72  # represented by a pair of two word IDs. The first word is at the
73  # "current" position, while the second word at the next position.
74  # At each iteration, the iteration count is incremented, which pushes
75  # forward the "current" position.
76  length = len(self.dataset)
77  if not self.repeat and self.iteration * self.batch_size >= length:
78  # If not self.repeat, this iterator stops at the end of the first
79  # epoch (i.e., when all words are visited once).
80  raise StopIteration
81  cur_words = self.get_words()
82  self._previous_epoch_detail = self.epoch_detail
83  self.iteration += 1
84  next_words = self.get_words()
85 
86  epoch = self.iteration * self.batch_size // length
87  self.is_new_epoch = self.epoch < epoch
88  if self.is_new_epoch:
89  self.epoch = epoch
90 
91  return list(zip(cur_words, next_words))
92 
93  @property
94  def epoch_detail(self):
95  # Floating point version of epoch.
96  return self.iteration * self.batch_size / len(self.dataset)
97 
98  @property
99  def previous_epoch_detail(self):
100  if self._previous_epoch_detail < 0:
101  return None
102  return self._previous_epoch_detail
103 
104  def get_words(self):
105  # It returns a list of current words.
106  return [self.dataset[(offset + self.iteration) % len(self.dataset)]
107  for offset in self.offsets]
108 
109  def serialize(self, serializer):
110  # It is important to serialize the state to be recovered on resume.
111  self.iteration = serializer('iteration', self.iteration)
112  self.epoch = serializer('epoch', self.epoch)
113  try:
114  self._previous_epoch_detail = serializer(
115  'previous_epoch_detail', self._previous_epoch_detail)
116  except KeyError:
117  # guess previous_epoch_detail for older version
118  self._previous_epoch_detail = self.epoch + \
119  (self.current_position - self.batch_size) / len(self.dataset)
120  if self.epoch_detail > 0:
121  self._previous_epoch_detail = max(
122  self._previous_epoch_detail, 0.)
123  else:
124  self._previous_epoch_detail = -1.

127 # Custom updater for truncated BackProp Through Time (BPTT)
128 class BPTTUpdater(training.StandardUpdater):
129 
130  def __init__(self, train_iter, optimizer, bprop_len, device):
131  super(BPTTUpdater, self).__init__(
132  train_iter, optimizer, device=device)
133  self.bprop_len = bprop_len
134 
135  # The core part of the update routine can be customized by overriding.
136  def update_core(self):
137  loss = 0
138  # When we pass one iterator and optimizer to StandardUpdater.__init__,
139  # they are automatically named 'main'.
140  train_iter = self.get_iterator('main')
141  optimizer = self.get_optimizer('main')
142 
143  # Progress the dataset iterator for bprop_len words at each iteration.
144  for i in range(self.bprop_len):
145  # Get the next batch (a list of tuples of two word IDs)
146  batch = train_iter.__next__()
147 
148  # Concatenate the word IDs to matrices and send them to the device
149  # self.converter does this job
150  # (it is chainer.dataset.concat_examples by default)
151  x, t = self.converter(batch, self.device)
152 
153  # Compute the loss at this time step and accumulate it
154  loss += optimizer.target(chainer.Variable(x), chainer.Variable(t))
155 
156  optimizer.target.cleargrads() # Clear the parameter gradients
157  loss.backward() # Backprop
158  loss.unchain_backward() # Truncate the graph
159  optimizer.update() # Update the parameters
160
